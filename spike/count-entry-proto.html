Paper → RPC · atomic contract

หน้านี้ไม่ใช่ form ธรรมดา · ต้องเข้าใจว่าแต่ละ row = 1 trigger fire ใน stock_counts + sm · batch ทั้งหมด wrap ใน single transaction เพื่อ atomicity · เอา pattern เดียวกับ stock.submit_delivery

client-side batch NEW

// aggregate per-row → one rpc call async function commitBatch(rows) { const payload = rows .filter(r => r.delta !== 0) .map(r => ({ item_id: r.item_id, counted_qty: r.paper_qty, note: `count-sheet ${weekOf} · ${slot}` })); await rpc("batch_count_adjust", { p_actor: actor, p_rows: payload // jsonb[] }); // atomic — all commit OR all fail }

server rpc_batch_count_adjust NEW · wrap existing

-- wraps existing rpc_count_adjust CREATE FUNCTION rpc_batch_count_adjust( p_actor text, p_rows jsonb ) RETURNS jsonb AS $$ DECLARE r jsonb; BEGIN FOR r IN SELECT * FROM jsonb_array_elements(p_rows) LOOP PERFORM rpc_count_adjust( p_actor, (r->>'item_id')::uuid, (r->>'counted_qty')::numeric, r->>'note' ); END LOOP; RETURN jsonb_build_object('ok', true, 'count', jsonb_array_length(p_rows)); END; $$ LANGUAGE plpgsql; -- fails any row → entire tx rolls back (pg default)
