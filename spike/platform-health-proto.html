SLO scan · regression watch contract

SLOs run ทุกชั่วโมง · watch run ทุกชั่วโมง · ผลเขียนเข้า platform_slo_history · dashboard query จาก view นี้เท่านั้น · ไม่เรียก live · lightweight · เปิด 50 ครั้ง/วันก็ไม่โหลด DB

-- regression watch #2 · catch_weight suspicious bags CREATE MATERIALIZED VIEW slo_catch_weight_watch AS SELECT now() AS scanned_at, count(*) FILTER (WHERE weight = 1.000) AS suspicious, count(*) AS total, 1.0 - (count(*) FILTER (WHERE weight = 1.000)::float / count(*)) AS pct_ok FROM meat_bags WHERE sku_code LIKE 'MT-%' AND created_at > now() - interval '30 days'; -- cron · pg_cron every hour SELECT cron.schedule('slo_scan', '0 * * * *', $$ REFRESH MATERIALIZED VIEW slo_catch_weight_watch; INSERT INTO platform_slo_history (slo_name, pct, scanned_at) SELECT 'catch_weight', pct_ok, now() FROM slo_catch_weight_watch; -- alert if pct_ok < 0.99 for > 12h IF (SELECT bool_and(pct < 0.99) FROM platform_slo_history WHERE slo_name='catch_weight' AND scanned_at > now() - interval '12 hours') THEN PERFORM notify_ops('SLO breach · catch_weight'); END IF; $$);
